NAME
	sp_value - get current cursor value

SYNOPSIS
	#include <sophia.h>

	const char *sp_value(void *cursor);

DESCRIPTION
	The sp_value(3) function is used to access currently positioned key
	value by a cursor.

EXAMPLE

	void *c = sp_cursor(db, SPGT, NULL, 0);
	if (c == NULL) {
		printf("error: %s\n", sp_error(db));
	}
	while (sp_fetch(c)) {
		printf("value: %s\n", sp_value(c));
		printf("value size: %d\n", sp_valuesize(c));
	}
	sp_destroy(c);


RETURN VALUE
	On success, sp_value(3) function returns value pointer;
	On error, sp_value(3) function returns NULL;
