NAME
	sp_rollback - discard a transaction changes

SYNOPSIS
	#include <sophia.h>

	int sp_rollback(void *db);

DESCRIPTION
	The sp_rollback(3) function is used to discard a changes of a multi-statement
	transaction. All modifications made during the transaction are not written to 
	the log file.

EXAMPLE

	int rc = sp_begin(db);
	if (rc == -1) {
		printf("error: %s\n", sp_error(env));
	}
	int key = 123;
	rc = sp_set(db, &key, sizeof(key), &key, sizeof(key));
	if (rc == -1) {
		printf("error: %s\n", sp_error(env));
	}
	rc = sp_rollback(db);
	if (rc == -1) {
		printf("error: %s\n", sp_error(env));
	}


RETURN VALUE
	On success, it returns 0.
	On error, it returns -1. See sp_error(3) for details.
