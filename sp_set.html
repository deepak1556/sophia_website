NAME
	sp_set - insert or replace a key-value pair

SYNOPSIS
	#include <sophia.h>

	int sp_set(void *db, const void *k, size_t ksize, const void *v, size_t vsize);

DESCRIPTION
	The sp_set(3) function is used to insert new or replace an existing
	key in a database.

EXAMPLE

	int key = 1234;
	char value[128];
	rc = sp_set(db, &key, sizeof(key), value, sizeof(value));
	if (rc == -1) {
		printf("error: %s\n", sp_error(db));
	}


RETURN VALUE
	On success, sp_set(3) functions returns 0;
	On error, it returns -1. See sp_error(3) for details.
