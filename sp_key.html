NAME
	sp_key - get current key

SYNOPSIS
	#include <sophia.h>

	const char *sp_key(void *cursor);

DESCRIPTION
	The sp_key(3) function is used to access currently positioned key
	by a cursor.

EXAMPLE

	void *c = sp_cursor(db, SPGT, NULL, 0);
	if (c == NULL) {
		printf("error: %s\n", sp_error(db));
	}
	while (sp_fetch(c)) {
		printf("key: %s\n", sp_key(c));
		printf("key size: %zu\n", sp_keysize(c));
	}
	sp_destroy(c);


RETURN VALUE
	On success, sp_key(3) function returns key pointer;
	On error, sp_key(3) function returns NULL;
